<? javascript
	var mem = new Array;
	var index = 0;
	function activate (index) {
		let msg = "/ITL/scene/cpINDEXm/cp penWidth 2;"
		msg += "/ITL/scene/cpINDEXm/cp watch mouseDown (";
		msg += "/ITL/scene/javascript run 'recall(INDEX, $(/ITL/scene/cp1m/cp get x), $(/ITL/scene/cp1m/cp get y), $(/ITL/scene/cp2m/cp get x), $(/ITL/scene/cp2m/cp get y), $(/ITL/scene/cp3m/cp get x), $(/ITL/scene/cp3m/cp get y), $(/ITL/scene/cp4m/cp get x), $(/ITL/scene/cp4m/cp get y))');"
		return msg.replace(/INDEX/g, index);
	}
	
	function recall (index, x1, y1, x2, y2, x3, y3, x4, y4) {
		let msg = "/ITL/scene/cpINDEXm/cp x XPOS;\n /ITL/scene/cpINDEXm/cp y YPOS;\n \
		/ITL/scene/faustINDEX/karplus/params/freq FREQ; \n \
		/ITL/scene/pINDEX tempo TEMPO;"
	
		let out = "";
		for (let i=0; i<4; i++) {
			let x =  mem[index-1][i*2];
			let y =  mem[index-1][i*2 + 1];
			let freq =  1320 - (y + 1)/2 * (1320 - 110);
			let tempo = 5 + (x+1)/2 * 315;
			out += msg.replace (/INDEX/g, i+1)
				.replace("XPOS", x)
				.replace("YPOS", y)
				.replace("FREQ", freq)
				.replace("TEMPO", tempo);
		}
		return out;
	}
	
	function store (x1, y1, x2, y2, x3, y3, x4, y4) {
		mem[index++] = [x1, y1, x2, y2, x3, y3, x4, y4];
		let msg = activate (index);
		if (index > 3) index = 0;
		return msg;
	}
?>
